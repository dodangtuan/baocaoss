create table Employees(
    employeeID int auto_increment,
	fullname varchar(255),
    age int,
    gender varchar(20),
    email varchar(255) ,
	phone varchar(20),
	salary decimal,
    managerID int null,
    deptID int null,
    isDelete boolean,
    primary key (employeeID),
 --    CONSTRAINT FK_managerID FOREIGN KEY (managerID) REFERENCES employees(employeeID),
    constraint fk_departmentID foreign key (deptID) references Departments(deptID)
);
